    pure function kepdh(dh, dn, sigma0, sqrta, a, r)

    implicit none

    real(wp),intent(in) :: dh      !! hyperbolic anomaly difference
    real(wp),intent(in) :: dn
    real(wp),intent(in) :: sigma0
    real(wp),intent(in) :: sqrta
    real(wp),intent(in) :: a
    real(wp),intent(in) :: r
    real(wp) :: kepdh

    kepdh = -dn - dh + sigma0 / sqrta * (cosh(dh) - one) + &
            (one - r / a) * sinh(dh)

    end function kepdh
